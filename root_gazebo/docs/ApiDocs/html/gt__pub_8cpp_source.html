191  void GtPub::GazeboLinkStatesCb
192  (const gazebo_msgs::LinkStates::ConstPtr& link_states){
193 
194  bool base_footprint_found = false;
195  tf2::Transform base_footprint_layout_tf;
196 
197  while(!base_footprint_found){
198  base_footprint_found =
199  this->GetBaseFootprintLayoutTf(link_states, &base_footprint_layout_tf);
200  if(!base_footprint_found){
201  ROS_WARN("base_footprint not found!!");
202  ros::Duration(0.1).sleep();
203  }
204  }
205 
206  static bool first = true;
207  if(first){
208  this->_gt_start_layout_tf = base_footprint_layout_tf;
209  this->PublishStaticTf(this->_layout,
210  "gt_start",
211  &this->_gt_start_layout_tf);
212  first = false;
213  }
214 
215  if(this->_publish_gt_path){
216  if( this->AddPosePath(&_gt_path,
217  &base_footprint_layout_tf, this->_layout,
218  this->_path_resolution) ){
219  this->_gt_path_pub.publish(this->_gt_path);
220  }
221  }
222 
223  tf2::Transform base_footprint_tf_tree_root_tf =
224  this->GetTf(this->_tf_tree_root, this->_base_footprint);
225 
226  tf2::Transform tf_tree_root_gt_start_tf =
227  (this->_gt_start_layout_tf.inverse()) *
228  (base_footprint_layout_tf) *
229  (base_footprint_tf_tree_root_tf.inverse());
230 
231  static int sequence_id = 0;
232  sequence_id++;
233  this->PublishTf("gt_start", this->_tf_tree_root,
234  sequence_id, &tf_tree_root_gt_start_tf);

237  bool GtPub::GetBaseFootprintLayoutTf
238  (const gazebo_msgs::LinkStates::ConstPtr& link_states,
239  tf2::Transform* tf){
240  int base_footprint_index;
241  bool base_footprint_found = false;
242  for(int i=0; i<link_states->name.size(); i++){
243  if(link_states->name.at(i) == this->_robot +
244  "::" +
245  this->_base_footprint){
246  base_footprint_index = i;
247  base_footprint_found = true;
248  break;
249  }
250  }
251 
252  if(!base_footprint_found){
253  return false;
254  }
255 
256  tf->setOrigin(tf2::Vector3(
257  link_states->pose.at(base_footprint_index).position.x,
258  link_states->pose.at(base_footprint_index).position.y,
259  link_states->pose.at(base_footprint_index).position.z));
260 
261  tf->setRotation(tf2::Quaternion(
262  link_states->pose.at(base_footprint_index).orientation.x,
263  link_states->pose.at(base_footprint_index).orientation.y,
264  link_states->pose.at(base_footprint_index).orientation.z,
265  link_states->pose.at(base_footprint_index).orientation.w));
266 
267  return true;
268  }

327  bool GtPub::AddPosePath(nav_msgs::Path* path,
328  tf2::Transform* tf, std::string frame_id,
329  double path_resolution){
330 
331  if(path->poses.size() != 0){
332  double distance = std::sqrt(
333  std::pow(tf->getOrigin().x() -
334  path->poses.front().pose.position.x, 2) +
335  std::pow(tf->getOrigin().y() -
336  path->poses.front().pose.position.y, 2) +
337  std::pow(tf->getOrigin().z() -
338  path->poses.front().pose.position.z, 2)
339  );
340 
341  if(distance < path_resolution){
342  return false;
343  }
344  }
345 
346  path->header.seq = 0;
347  path->header.stamp = ros::Time::now();
348  path->header.frame_id = frame_id;
349 
350  geometry_msgs::PoseStamped pose;
351  pose.header.seq = 0;
352  pose.header.stamp = ros::Time::now();
353  pose.header.frame_id = frame_id;
354  tf2::toMsg(*tf, pose.pose);
355 
356  path->poses.emplace_back(pose);
357  return true;
358  }
